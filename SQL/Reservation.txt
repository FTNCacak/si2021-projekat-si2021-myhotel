CREATE TABLE [dbo].[Reservation] (
    [Id]         INT        IDENTITY (1, 1) NOT NULL,
    [Payment]    FLOAT (53) NOT NULL,
    [Begin Date] DATETIME       NOT NULL,
    [End Date]   DATETIME       NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC),
    CONSTRAINT [FK_Guest] FOREIGN KEY ([Id]) REFERENCES [dbo].[Guest] ([Id]),
    CONSTRAINT [FK_Appartment] FOREIGN KEY ([Id]) REFERENCES [dbo].[Appartment] ([Id])
);
